import numpy as np
import pandas as pd
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score

# Step 1: Generate 100 random values in the range [0,1]
x_values = np.random.rand(100)

# Step 2: Label the first 50 points
labels = np.array(["Class1" if x <= 0.5 else "Class2" for x in x_values[:50]])

# Step 3 Create Training Data 
X_train=x_values[:50].reshape(-1,1)
y_train=labels

# Step 4 Create DataFrame for Training Data 
df1=pd.DataFrame({"Points":[f"x{i+1}" for i in range(50)],"Value":x_values[:50],"Label":labels})
print(df1)

# Step 5 Create Testing Data 
X_test=x_values[50:].reshape(-1,1)

#Step 6 Create the actual Label for the remaining 50 Points
true_labels = np.array(["Class1" if x <= 0.5 else "Class2" for x in x_values[50:]])

# Step 7: Perform KNN classification for different values of k
k_values = [1, 2, 3, 4, 5, 20, 30]
results = {}
accuracies = {}

for k in k_values:
    knn = KNeighborsClassifier(n_neighbors=k)
    knn.fit(X_train, y_train)
    predictions = knn.predict(X_test)
    results[k] = predictions
    # Calculate accuracy
    accuracy = accuracy_score(true_labels, predictions) * 100
    accuracies[k] = accuracy
    print(f"Accuracy for k={k}: {accuracy:.2f}%")

# Step 8 Create a DataFrame for remaining 50 points for k=2 with Predicted label and actual label 
df2=pd.DataFrame({"Points":[f"x{i+1}" for i in range(50,100)],"actual labels":true_labels,"Predicted Labels": results[2]})
print(df2)
